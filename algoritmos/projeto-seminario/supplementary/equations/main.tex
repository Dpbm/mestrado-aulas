\begin{algorithm}
\caption{Hyper-Par}
\begin{algorithmic}

\State $B \gets \infty$ \Comment{Melhor custo}
\State $BG \gets \{\}$ \Comment{Melhor árvore}
\Statex

\Function{Cost}{Node}
    \If {Node is leaf}
        \State \Return $0$
    \EndIf
    \Statex

    \State $c \gets 2^{nindexes}$
    \State $\text{total} \gets c$
    \Statex

    \For{child in node.children}
        \State $\text{total} \gets \text{Cost(child)}$
    \EndFor
    \Statex

    \State \Return total
\EndFunction
\Statex

\Function{Opt}{$V$}
    \State $\text{out\_nodes} \gets \{\}$
    \State $\text{nodes} \gets \text{every item in $V$ as Leaf}$

    \While{ $len(nodes) > 1$}

        \State $\text{pairs} \gets \{\}$
        \State $\text{costs} \gets \{\}$
        \Statex
        
        \For{each pair (i, j) in nodes}
            \State $\text{cost} \gets \text{size($\text{node}_i$)} - \alpha (\text{size($\text{node}_i$)} + \text{size($\text{node}_j$)})$

            \State pairs.append((i, j))
            \State costs.append(cost)
        \EndFor
        \Statex
        
        \State $\text{probs} \gets \{\}$
        \State $\text{Z} \gets 0$
        \For{cost in costs}
            \State $p \gets \exp{(-cost / \tau)}$
            \State probs.append(p)
            \State $Z \gets Z + p$
        \EndFor
        \Statex

        
        \For{ $k$ in range(len(probs))}
            \State $probs_k \gets \frac{probs_k}{Z} $
        \EndFor

        \State $\text{i, j} \gets \text{random\_choice(pairs, probs)}$
        
        \State $\text{new\_node} \gets \text{Node($\text{nodes}_i$, $\text{nodes}_j$)}$

        \State $\text{remove nodes}_i\text{, nodes}_j\text{ from nodes}$
        \State out\_nodes.append(new\_node)
    \EndWhile

    \State \Return out\_nodes
\EndFunction
\Statex

\Function{Partition}{$V$,cut,$\epsilon$,$k$}
    \If{$|V| \le 1$}
        \State \Return Leaf($V$)
    \EndIf

    \If{$|V| \le \text{cut}$}
        \State \Return Opt($V$)
    \EndIf
    \Statex

    \State $\text{parts} \gets \text{KaHyPar($V$, $\epsilon$, $k$)}$   
    \State $\text{children} \gets \{\}$
    \Statex

    \For{each $V_i$ in parts}
        \State children.append(Partition($V_i$, cut, $\epsilon$, $k$))
    \EndFor
    
    \State \Return Node($V$, children)
\EndFunction
\Statex

\For{ $\text{n combs. de } k, \epsilon,\text{cut}$ }

\State $V \gets \text{Partition(${G}$)}$

\State $\text{cost} \gets \text{Cost($V$)}$
\If{$\text{cost} \lt B$}
    \State $B \gets \text{cost}$
    \State $BG \gets V$
\EndIf

\EndFor


\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
\SetKwFunction{FExpand}{expandBFS}
\SetKwFunction{FContract}{contract}
\SetKwFunction{FWeight}{weight}
\SetKwFunction{FLimit}{calculateLimit}
\SetKwInOut{Input}{input}\SetKwInOut{Output}{output}

\Input{Hypergraph $H = (V, E)$, Current Partition $(V_0, V_1)$, Capacity parameter $\alpha$}
\Output{Sub-hypergraph $H'$ for FlowCutter with source $s$ and target $t$}

\BlankLine
\tcp{1. Identify initial border vertices}
$B_0 \leftarrow \{v \in V_0 \mid \exists e \in E : v \in e \land e \cap V_1 \neq \emptyset\}$\;
$B_1 \leftarrow \{v \in V_1 \mid \exists e \in E : v \in e \land e \cap V_0 \neq \emptyset\}$\;

\BlankLine
\tcp{2. Expand to find the Movable Set $M$}
$L_0 \leftarrow \FLimit(\alpha, V_0)$ \tcp*{Weight limit for block 0 expansion}
$L_1 \leftarrow \FLimit(\alpha, V_1)$ \tcp*{Weight limit for block 1 expansion}

$M_0 \leftarrow \FExpand(B_0, V_0, L_0)$\;
$M_1 \leftarrow \FExpand(B_1, V_1, L_1)$\;
$M \leftarrow M_0 \cup M_1$\;

\BlankLine
\tcp{3. Contract stable vertices into source and target}
$s \leftarrow \FContract(V_0 \setminus M)$\;
$t \leftarrow \FContract(V_1 \setminus M)$\;
$V_{active} \leftarrow M \cup \{s, t\}$\;

\BlankLine
\tcp{4. Filter hyperedges for the flow network}
$E' \leftarrow \emptyset$\;
\ForEach{$e \in E$}{
    $e' \leftarrow e \cap V_{active}$\;
    \If{$|e'| \ge 2$ \textbf{and} $e$ is not connected to other blocks}{
        $E' \leftarrow E' \cup \{e'\}$\;
    }
}

\Return $H' = (V_{active}, E')$ with terminals $(s, t)$\;

\label{alg:flowcutter_setup}
\end{algorithm}
