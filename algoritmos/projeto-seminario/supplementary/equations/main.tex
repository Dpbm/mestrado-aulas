\begin{algorithm}
\caption{Hyper-Par}
\begin{algorithmic}

\State $B \gets \infty$ \Comment{Melhor custo}
\State $BG \gets \{\}$ \Comment{Melhor Ăˇrvore}
\Statex

\Function{Cost}{Node}
    \If {Node is leaf}
        \State \Return $0$
    \EndIf
    \Statex

    \State $c \gets 2^{indexes}$
    \State $\text{total} \gets c$
    \Statex

    \For{child in node.children}
        \State $\text{total} \gets \text{total} + \text{Cost(child)}$
    \EndFor
    \Statex

    \State \Return total
\EndFunction
\Statex

\Function{Opt}{$V$}
    \State $\text{outNodes} \gets \{\}$
    \State $\text{nodes} \gets \text{every item in $V$ as Leaf}$

    \While{ $len(nodes) > 1$}

        \State $\text{pairs} \gets \{\}$
        \State $\text{costs} \gets \{\}$
        \Statex
        
        \For{each pair (i, j) in nodes}
            \State $\text{cost} \gets \text{size($\text{node}_i$)} - \alpha (\text{size($\text{node}_i$)} + \text{size($\text{node}_j$)})$

            \State pairs.append((i, j))
            \State costs.append(cost)
        \EndFor
        \Statex
        
        \State $\text{probs} \gets \{\}$
        \State $\text{Z} \gets 0$
        \For{cost in costs}
            \State $p \gets \exp{(-cost / \tau)}$
            \State probs.append(p)
            \State $Z \gets Z + p$
        \EndFor
        \Statex

        
        \For{ $k$ in range(len(probs))}
            \State $probs_k \gets probs_k / Z $
        \EndFor

        \State $\text{i, j} \gets \text{randomChoice(pairs, probs)}$
        
        \State $\text{newNode} \gets \text{Node($\text{nodes}_i$, $\text{nodes}_j$)}$

        \State $\text{remove nodes}_i\text{, nodes}_j\text{ from nodes}$
        \State outNodes.append(newNode)
    \EndWhile

    \State \Return outNodes
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Coarsening}
\begin{algorithmic}
    \State nodes $\gets $ shuffle(G.nodes)
    \State matched $\gets \{\}$
    \Statex
    \For{node in nodes}
        \If{node in matched}
            \State continue
        \EndIf
        \State bestNeighbor $\gets$ null
        \State maxRating $\gets -\infty$
        \Statex

        \State neighbors $\gets$ neighborsLocalClusters(node)
        \For{neighbor in neighbors}
            \If{neighbor $==$ node $||$ neighbor in matched $||$ $\text{weight}_\text{node} + \text{weight}_\text{neighbor} > \text{maxVertexWeight}$ }
                \State continue
            \EndIf
            \Statex
            

            \State $r \gets \text{rate}(\text{node},\text{neighbor})$
            
            \If{$r > \text{maxRating}$}
                \State $\text{maxRating} \gets r$
                \State $\text{bestNeighbor} \gets \text{neighbor}$
            \EndIf
        \EndFor
        \Statex

        \If{bestNeighbor}
            \State contract(node, bestNeighbor)
            \State matched.add(node, bestNeighbor)
        \EndIf
    
    \EndFor
    \Statex

\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{functions}
\begin{algorithmic}

    \Function{neighborsLocalClusters}{node}
        \State neighbors $\gets \{\}$
        \For{edge in incidentEdges$_\text{node}$}
            \If{len(edge.vertices) $\le$ threshold}
                \Statex
                \For{neighbor in edge.vertices}
                    \If{neighbor $\ne$ node}
                        \State neighbors.add(neighbor)
                    \EndIf
                \EndFor
                \Statex
            \EndIf
        \EndFor
        \Statex
        \State \Return neighbors
    \EndFunction
    \Statex

    \Function{rate}{node, neighbor}
        \State s $\gets \text{incidentEdges}_\text{node} \cap \text{incidentEdges}_\text{neighbor}$
        \State \Return $\sum_{e}^{s}{1/\text{len(e.vertices)} - 1}$
    \EndFunction
    \Statex

    \Function{contract}{node, neighbor}
       \State node.weight $\gets$ node.weight $+$ neighbor.weight

       \For{edge in incidentEdges$_\text{neighbor}$}
            \State edge.removeVertex(neighbor)
            \If{! node in edge.vertices}
                \State edge.addVertex(node)
                \State node.addIncidentEdge(edge)
            \EndIf
            \Statex

            \If{len(edge.vertices) $< 2$}
                \State removeEdge(edge)
            \EndIf
       \EndFor
        \State removeVertex(neighbor)
    \EndFunction

\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{RB}
\begin{algorithmic}
    \Function{recursiveBipartition}{$G$, $k$}
        \If{$k \le 1$}:
            \State \Return $\{\}$
        \EndIf
    
        \State $V_0, V_1 \gets$  2wayPartition($G$)
        \State $G_0 \gets$ new graph with only $V_0$
        \State $G_1 \gets$ new graph with only $V_1$

        \State left $\gets$ recursiveBipartition($G_0$, $k/2$)
        \State right $\gets$ recursiveBipartition($G_1$, $k/2$)
        \State \Return $\text{left} + \text{right}$
    \EndFunction
    \Statex

    \Function{2wayPartition}{$G$}
        \State hierarchy $\gets \{G\}$
        \State currentG $\gets G$

            
        \While{currentG.numVertices $>$ threshold} 
            \State newG $\gets$ coarse(currentG)
            \State hierarchy.append(newG)
            \State currentG $\gets$ newG
        \EndWhile
        \Statex
    
        \State cut $\gets$ partition(currentG)
        \Statex
        
        \While{hierarchy}
            \State coarsedG $\gets$ hierarchy.pop()
            \State cut $\gets$ projectCut(cut, coarsedG)
        \EndWhile

        \State \Return cut
    \EndFunction
    \Statex

    \Function{projectCut}{$cut$, $G$}
        \State newPartition $\gets \{\}$
    
        \For{node in G.nodes}
            \State SU $\gets$ node.parent 
            \State newPartition$_\text{node} \gets \text{cut}_\text{SU}$
        \EndFor
        \State \Return newPartition
    \EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithmic}
    \Function{partition}{$G$}
        \State bestSize $\gets \infty$
        \State bestPartition $\gets$ null
        \Statex

        \For{attempt in range(attempts)}

            \State $rand_0, rand_1 \gets $ Two random nodes from $G$
            \State $block_0 \gets \{rand_0\}$
            \State $block_1 \gets \{rand_1\}$
            \Statex
            
            \State $candidates_0 \gets \text{PriorityQueue}(rand_0.\text{neighbors})$
            \State $candidates_1 \gets \text{PriorityQueue}(rand_1.\text{neighbors})$
            \State $\text{targetWeight} \gets \text{G.totalWeight}/2$
            \Statex
            
            \While{$len(block_0) + len(block_1) < \text{G.vertices}$}
                
                \State currentBlock $\gets$ Current Smaller block
                \State currentQueue $\gets$ Queue relative to the current block
                \Statex
                
                \If{! currentQueue.empty}
                    \State maxNode $\gets$ currentQueue.popMaxGainNode()
                    
                    \If{! maxNode in $block_0$ $\&\&$ ! maxNode in $block_1$}
                        \State currentBlock.add(maxNode)
                        \Statex
                        
                        \For{neighbor in maxNode.neighbors}
                            \If{! neighbor in $block_0$ $\&\&$ ! neighbor in $block_1$}
                                \State currentQueue.insert(neighbor)
                            \EndIf
                        \EndFor
                        \Statex
                        
                    \EndIf
                \Else
                    \State currentBlock.add(G.getRandomUnassignedNode())
                \EndIf
            \EndWhile
            \Statex

            \State partition $\gets (block_0, block_1)$ 
        
            \If{partition.cutSize $<$ bestSize}
                \State bestSize $\gets$ partition.cutSize
                \State bestPartition $\gets$ partition
            \EndIf
    
        \EndFor
    
    
        \State \Return bestPartition
    \EndFunction
\end{algorithmic}
\end{algorithm}
